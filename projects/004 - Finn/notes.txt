FROM RLS.vw_COVID19 
where 
Not ((
	(GroupDescription = 'Confirmed' AND SubGroupDescription != 'Negative') OR
	(GroupDescription = 'Tested' AND SubGroupDescription = 'Positive')
))
and	DeathWithin28Days != 'N'
